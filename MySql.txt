Thay đổi thông tin đăng nhập MySQL trong application.properties

CREATE DATABASE apartment_management;
USE apartment_management;

-- Bảng users (quản trị viên và cư dân)
CREATE TABLE users (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    password VARCHAR(100) NOT NULL,
    role ENUM('ADMIN', 'RESIDENT') NOT NULL,
    avatar VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Bảng surveys (phiếu khảo sát)
CREATE TABLE surveys (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(255) NOT NULL,
    description TEXT,
    created_by BIGINT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    status ENUM('OPEN', 'CLOSED') DEFAULT 'OPEN',
    FOREIGN KEY (created_by) REFERENCES users(id)
);

-- Bảng survey_questions (câu hỏi trong phiếu khảo sát)
CREATE TABLE survey_questions (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    survey_id BIGINT NOT NULL,
    question_text TEXT NOT NULL,
    question_type ENUM('TEXT', 'MULTIPLE_CHOICE', 'RATING') NOT NULL,
    options JSON, -- Lưu danh sách lựa chọn nếu là MULTIPLE_CHOICE
    FOREIGN KEY (survey_id) REFERENCES surveys(id)
);

-- Bảng survey_responses (câu trả lời của cư dân)
CREATE TABLE survey_responses (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    survey_id BIGINT NOT NULL,
    question_id BIGINT NOT NULL,
    user_id BIGINT NOT NULL,
    response_text TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (survey_id) REFERENCES surveys(id),
    FOREIGN KEY (question_id) REFERENCES survey_questions(id),
    FOREIGN KEY (user_id) REFERENCES users(id)
);
